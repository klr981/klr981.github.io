  Vector4? uColor,

      ...uCenter != null ? uCenter.storage : <double>[0, 0],

      ...uMaxRadius != null ? <double>[uMaxRadius] : <double>[0],

      ...uNoisePhase != null ? <double>[uNoisePhase] : <double>[0],

    ]),
  );
